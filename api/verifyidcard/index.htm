function checkIdCode(code) {
 if (code == null || code.length != 18) return false;
 var factor = [7, 9, 10, 5, 8, 4, 2, 1, 6, 3, 7, 9, 10, 5, 8, 4, 2];
 var random = "10X98765432".split("");
 var total = 0;
 for (var i = 0; i < 17; i++) {
  total += code.charAt(i) * factor[i];
 }
 return random[total % 11] == code[17];
}
